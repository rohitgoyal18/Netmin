#!/bin/bash
clear
echo "1) Create a new partition."
echo "2) Mount a filesystem."
echo "3) Unmount a filesystem."
echo "4) Create a filesystem. "
echo "5) Craete a NFS export."
echo "6) Create a NFS import."
echo "Enter your choice(1-6) : "
read choice

case $choice in
1)	./partition
	;;
2)	./mount
	;;
3)	./umount
	;;
4)	./mkfs
	;;
5)	./nfsexport
	;;
6)	./nfsimport
	;;
	
*)	echo "Invalid option"
	;;

esac
